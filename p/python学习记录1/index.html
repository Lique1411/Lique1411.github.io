#导入模块
import argparse

## 定义参数设置函数
def parse_arguments():
    # 设置参数解析器
    parser = argparse.ArgumentParser(description='护花米草遥感识别处理程序')
    ## 添加必填参数与非必填参数
    #必需的参数
    parser.add_argument('-i', '--input', required=True,
                        help='输入数据文件夹路径（包含nc文件）')
    parser.add_argument('-w', '--word', required=True,
                        help='Word模板文件路径（.docx文件）')
    parser.add_argument('-o', '--output', required=True,
                        help='输出结果保存路径')

    #非必填，带默认值的参数
    parser.add_argument('--threshold', type=float, default=0.5, help='阈值')
    
    # 解析参数
    args = parser.parse_args()

    #返回参数
    return args

## 调用函数
def main():
    args = parse_arguments()
    print(f"{Fore.MAGENTA}输入路径为：{args.input_path}{Style.RESET_ALL}")
